73/// Errors that can occur when parsing or loading data.
74#[derive(Debug, Error)]
75pub enum DataParseError {
76    /// A std::io::Error
77    #[error(transparent)]
78    Io(#[from] std::io::Error),
79    /// a serde_json::Error
80    #[error(transparent)]
81    Serde(#[from] serde_json::Error),
82}

84/// Represents the json in its entirety
85#[derive(Debug, Clone, Serialize, Deserialize)]
86struct LoadedCards {
87    /// Has all information related to versioning and game mode
88    metadata: LoadedCardsMetadata,
89    /// Has the asset deck, liability deck and market and events deck
90    deck_list: DeckList,
91}
92
93/// Card metadata
94#[derive(Debug, Clone, Serialize, Deserialize)]
95struct LoadedCardsMetadata {
96    /// The json version
97    version: String,
98    /// The name of the gamemode
99    gamemode: String,
100}

102/// The list of decks in the json: assets, liabilities and markets/events.
103#[derive(Debug, Clone, Serialize, Deserialize)]
104struct DeckList {
105    /// List of all asset cards in the game
106    asset_deck: Deck<AssetCard>,
107    /// List of all liability cards in the game
108    liability_deck: Deck<LiabilityCard>,
109    /// List of all market/event cards in the game
110    market_events_deck: Deck<MarketEventCard>,
111}

113/// Representation of an asset in the json
114#[derive(Debug, Clone, Serialize, Deserialize)]
115struct AssetCard {
116    /// Title of the card
117    title: String,
118    /// Color of the card
119    color: Color,
120    /// Gold value of the card
121    gold_value: u8,
122    /// Silver value of the card
123    silver_value: u8,
124    /// Amount of times the card appears in the deck
125    copies: u8,
126    /// Url containing the relative location of the card in the assets folder
127    card_image_url: String,
128    /// Possible ability of the card
129    ability: Option<AssetPowerup>,
130}

132/// Representation of a liability card as it appears in the json
133#[derive(Debug, Clone, Serialize, Deserialize)]
134struct LiabilityCard {
135    /// Type of liability: Trade Credit, Bank Loan or Bonds
136    liability_type: LiabilityType,
137    /// Gold value of the liability
138    gold_value: u8,
139    /// Amount of times the card appears in the deck
140    copies: u8,
141    /// Url containing the relative location of the card in the assets folder
142    card_image_url: String,
143}

145/// Representation of either a market or event card as it appears in the json
146#[derive(Debug, Clone, Serialize, Deserialize)]
147struct MarketEventCard {
148    /// Title of the card
149    pub title: String,
150    /// Amount of times the card appears in the deck
151    pub copies: u32,
152    /// Url containing the relative location of the card in the assets folder
153    pub card_image_url: String,
154
155    /// Representation of either the market or event specific fields, because both are in this list
156    #[serde(flatten)]
157    pub details: MarketEventDetails,
158}
159
160/// Enum representing the fields only found on either a market or an event
161#[derive(Debug, Clone, Serialize, Deserialize)]
162#[serde(untagged)]
163enum MarketEventDetails {
164    MarketStatus { market_status: MarketStatusCard },
165    Event { event: EventCard },
166}
167
168/// Card representing the non-shared fields with event as it appears in the json
169#[derive(Debug, Clone, Serialize, Deserialize)]
170struct MarketStatusCard {
171    pub rfr: u8,
172    pub mrp: u8,
173    #[serde(rename = "Yellow", default)]
174    pub yellow: MarketCondition,
175    #[serde(rename = "Blue", default)]
176    pub blue: MarketCondition,
177    #[serde(rename = "Green", default)]
178    pub green: MarketCondition,
179    #[serde(rename = "Purple", default)]
180    pub purple: MarketCondition,
181    #[serde(rename = "Red", default)]
182    pub red: MarketCondition,
183}

185#[derive(Debug, Clone, Serialize, Deserialize)]
186struct EventCard {
187    pub description: String,
188    pub effect: String,
189}

191/// Overarching struct which contains the asset deck, the liability deck and the market/event deck.
192/// This can be used by the game or anything else that might want to get all cards.
193#[derive(Debug, Clone)]
194pub struct GameData {
195    /// Deck containing all assets
196    pub assets: Deck<Asset>,
197    /// Deck containing all liabilities
198    pub liabilities: Deck<Liability>,
199    /// Deck containing all markets and events
200    pub market_deck: Deck<Either<Market, Event>>,
201}
202
203impl GameData {
204    /// Tries loading a json at `cards_json_path`. It reads the file to string and tries to parse
205    /// that string into a [`GameData`] struct using `serde_json`.
206    pub fn new<P: AsRef<Path>>(cards_json_path: P) -> Result<GameData, DataParseError> {
207        let json = read_to_string(cards_json_path)?;
208
209        let cards = serde_json::from_str::<LoadedCards>(&json)?;
210
211        Ok(Self::from(cards))
212    }
213
214    /// Shuffles each individual deck.
215    #[cfg(feature = "shuffle")]
216    pub fn shuffle_all(&mut self) {
217        self.assets.shuffle();
218        self.liabilities.shuffle();
219        self.market_deck.shuffle();
220    }
221}

223impl From<Deck<AssetCard>> for Deck<Asset> {
224    fn from(cards: Deck<AssetCard>) -> Self {
225        let image_back_url = cards.image_back_url.clone();
226        let deck = cards
227            .deck
228            .into_iter()
229            .flat_map(|c| {
230                // keep borrow checker happy about moving an Arc into each Asset
231                let image_back_url = image_back_url.clone();
232
233                (0..c.copies).map(move |_| Asset {
234                    title: c.title.clone(),
235                    gold_value: c.gold_value,
236                    silver_value: c.silver_value,
237                    color: c.color,
238                    ability: c.ability,
239                    image_front_url: c.card_image_url.clone(),
240                    image_back_url: image_back_url.clone(),
241                })
242            })
243            .collect::<Vec<_>>();
244
245        Deck::new_with_url(deck, &image_back_url)
246    }
247}

249impl From<Deck<LiabilityCard>> for Deck<Liability> {
250    fn from(cards: Deck<LiabilityCard>) -> Self {
251        let image_back_url = cards.image_back_url;
252        let deck = cards
253            .deck
254            .into_iter()
255            .flat_map(|c| {
256                // keep borrow checker happy about moving an Arc into each Liability
257                let image_back_url = image_back_url.clone();
258
259                (0..c.copies).map(move |_| Liability {
260                    value: c.gold_value,
261                    rfr_type: c.liability_type,
262                    image_front_url: c.card_image_url.clone(),
263                    image_back_url: image_back_url.clone(),
264                })
265            })
266            .collect::<Vec<_>>();
267
268        Deck::new_with_url(deck, &image_back_url)
269    }
270}

272impl From<Deck<MarketEventCard>> for Deck<Either<Market, Event>> {
273    fn from(cards: Deck<MarketEventCard>) -> Self {
274        let image_back_url = cards.image_back_url;
275        let deck = cards
276            .deck
277            .into_iter()
278            .flat_map(|c| {
279                (0..c.copies).map(move |_| match c.details.clone() {
280                    MarketEventDetails::MarketStatus { market_status } => Either::Left(Market {
281                        title: c.title.clone(),
282                        rfr: market_status.rfr,
283                        mrp: market_status.mrp,
284                        red: market_status.red,
285                        green: market_status.green,
286                        blue: market_status.blue,
287                        yellow: market_status.yellow,
288                        purple: market_status.purple,
289                    }),
290                    MarketEventDetails::Event { event } => Either::Right(Event {
291                        title: c.title.clone(),
292                        description: event.description.clone(),
293                        plus_gold: HashSet::new(),
294                        minus_gold: HashSet::new(),
295                        skip_turn: None,
296                    }),
297                })
298            })
299            .collect::<Vec<_>>();
300
301        Deck::new_with_url(deck, &image_back_url)
302    }
303}

305impl From<LoadedCards> for GameData {
306    fn from(cards: LoadedCards) -> Self {
307        GameData {
308            assets: cards.deck_list.asset_deck.into(),
309            liabilities: cards.deck_list.liability_deck.into(),
310            market_deck: cards.deck_list.market_events_deck.into(),
311        }
312    }
313}
