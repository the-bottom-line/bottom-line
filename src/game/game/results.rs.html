5/// State containing all information related to the results state of the game. In the resuts stage,
6/// players can see their scores.
7#[derive(Debug, Clone, PartialEq)]
8pub struct Results {
9    pub(super) players: Players<ResultsPlayer>,
10    // TODO: implement events
11    pub(super) final_events: Vec<Event>,
12}
13
14impl Results {
15    /// Get a reference to a [`ResultsPlayer`] based on a specific `PlayerId`. Note that the players
16    /// are in order, so id 0 refers to the player at index 0 and so on.
17    /// See [`Players::player`] for further information
18    pub fn player(&self, id: PlayerId) -> Result<&ResultsPlayer, GameError> {
19        self.players.player(id)
20    }
21
22    /// Get a reference to a [`ResultsPlayer`] based on a specific `name`.
23    pub fn player_by_name(&self, name: &str) -> Result<&ResultsPlayer, GameError> {
24        self.players()
25            .iter()
26            .find(|p| p.name() == name)
27            .ok_or_else(|| GameError::InvalidPlayerName(name.to_owned()))
28    }
29
30    /// Gets a slice of all players in the lobby.
31    /// See [`Players::players`] for further information
32    pub fn players(&self) -> &[ResultsPlayer] {
33        self.players.players()
34    }
35
36    /// Returns a list of [`PlayerScore`], which contains the player id as well as their final
37    /// score.
38    pub fn player_scores(&self) -> Vec<PlayerScore> {
39        self.players()
40            .iter()
41            .map(|p| PlayerScore::new(p.id(), p.name(), p.score()))
42            .collect()
43    }
44
45    /// Gets the [`PlayerInfo`] for each player, excluding the player that has the same id as `id`.
46    pub fn player_info(&self, id: PlayerId) -> Vec<PlayerInfo> {
47        self.players()
48            .iter()
49            .filter(|p| p.id() != id)
50            .map(Into::into)
51            .collect()
52    }
53
54    /// Gets the list of events that happened over the course of the game
55    pub fn final_events(&self) -> &[Event] {
56        &self.final_events
57    }
58
59    /// Increases one of the market conditions of a certain color for player with `id`. This means
60    /// that minus is turned into zero and zero is turned into plus. Returns the resulting market.
61    pub fn toggle_minus_into_plus(
62        &mut self,
63        id: PlayerId,
64        color: Color,
65    ) -> Result<Market, GameError> {
66        let player = self.players.player_mut(id)?;
67        let market = player.toggle_minus_into_plus(color)?;
68
69        Ok(market.to_owned())
70    }
71
72    /// Toggles the [`AssetPowerup::SilverIntoGold`] asset ability for a particular player.
73    pub fn toggle_silver_into_gold(
74        &mut self,
75        id: PlayerId,
76        asset_idx: usize,
77    ) -> Result<ToggleSilverIntoGold, GameError> {
78        let player = self.players.player_mut(id)?;
79        let data = player.toggle_silver_into_gold(asset_idx)?;
80
81        Ok(data)
82    }
83
84    /// Toggles the [`AssetPowerup::CountAsAnyColor`] asset ability for a particular player.
85    pub fn toggle_change_asset_color(
86        &mut self,
87        id: PlayerId,
88        asset_idx: usize,
89        color: Color,
90    ) -> Result<ToggleChangeAssetColor, GameError> {
91        let player = self.players.player_mut(id)?;
92        let data = player.toggle_change_asset_color(asset_idx, color)?;
93
94        Ok(data)
95    }
96
97    /// Asset abilities are toggleable by default. This function confirms the current configuration
98    /// for this particular player, after which they cannot toggle this particular index anymore.
99    pub fn confirm_asset_ability(
100        &mut self,
101        id: PlayerId,
102        asset_idx: usize,
103    ) -> Result<(), GameError> {
104        let player = self.players.player_mut(id)?;
105        player.confirm_asset_ability(asset_idx)
106    }
107}

109/// Representation of a player's final score, which contains their id as well as their score.
110///
111/// # Examples
112///
113/// ```
114/// # use game::{game::PlayerScore, player::PlayerId};
115/// let score = PlayerScore::new(PlayerId(0), "oxey", 10.0);
116/// assert_eq!(score.id(), PlayerId(0));
117/// assert_eq!(score.name(), "oxey");
118/// assert_eq!(score.score(), 10.0);
119/// ```
120#[cfg_attr(feature = "ts", derive(TS))]
121#[cfg_attr(feature = "ts", ts(export_to = crate::SHARED_TS_DIR))]
122#[derive(Clone, Debug, Serialize, Deserialize)]
123pub struct PlayerScore {
124    id: PlayerId,
125    name: String,
126    score: f64,
127}
128
129impl PlayerScore {
130    /// Constructs a new [`PlayerScore`]
131    ///
132    /// # Examples
133    ///
134    /// ```
135    /// # use game::{game::PlayerScore, player::PlayerId};
136    /// let score = PlayerScore::new(PlayerId(0), "oxey", 10.0);
137    /// ```
138    pub fn new(id: PlayerId, name: &str, score: f64) -> Self {
139        let name = name.to_owned();
140        Self { id, name, score }
141    }
142
143    /// Gets a [`PlayerScore`]'s `id` field.
144    ///
145    /// # Examples
146    ///
147    /// ```
148    /// # use game::{game::PlayerScore, player::PlayerId};
149    /// let score = PlayerScore::new(PlayerId(0), "oxey", 10.0);
150    /// assert_eq!(score.id(), PlayerId(0));
151    /// ```
152    pub fn id(&self) -> PlayerId {
153        self.id
154    }
155
156    /// Gets a [`PlayerScore`]'s `name` field.
157    ///
158    /// # Examples
159    ///
160    /// ```
161    /// # use game::{game::PlayerScore, player::PlayerId};
162    /// let score = PlayerScore::new(PlayerId(0), "oxey", 10.0);
163    /// assert_eq!(score.name(), "oxey");
164    /// ```
165    pub fn name(&self) -> &str {
166        &self.name
167    }
168
169    /// Gets a [`PlayerScore`]'s `score` field.
170    ///
171    /// # Examples
172    ///
173    /// ```
174    /// # use game::{game::PlayerScore, player::PlayerId};
175    /// let score = PlayerScore::new(PlayerId(0), "oxey", 10.0);
176    /// assert_eq!(score.score(), 10.0);
177    /// ```
178    pub fn score(&self) -> f64 {
179        self.score
180    }
181}
