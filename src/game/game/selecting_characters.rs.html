7/// State containing all information related to the selecting characters state of the game. In the
8/// selecting characters stage, players select a character one by one until everyone has selected
9/// a character, after which a round starts.
10#[derive(Debug, Clone, PartialEq)]
11pub struct SelectingCharacters {
12    pub(super) players: Players<SelectingCharactersPlayer>,
13    pub(super) characters: ObtainingCharacters,
14    pub(super) assets: Deck<Asset>,
15    pub(super) liabilities: Deck<Liability>,
16    pub(super) markets: Deck<Either<Market, Event>>,
17    pub(super) chairman: PlayerId,
18    pub(super) current_market: Market,
19    pub(super) current_events: Vec<Event>,
20}
21
22impl SelectingCharacters {
23    /// Get a reference to a [`SelectingCharactersPlayer`] based on a specific `PlayerId`. Note
24    /// that the players are in order, so id 0 refers to the player at index 0 and so on.
25    /// See [`Players::player`] for further information
26    pub fn player(&self, id: PlayerId) -> Result<&SelectingCharactersPlayer, GameError> {
27        self.players.player(id)
28    }
29
30    /// Get a reference to a [`SelectingCharactersPlayer`] based on a specific `name`. Note
31    /// that the players are in order, so id 0 refers to the player at index 0 and so on.
32    pub fn player_by_name(&self, name: &str) -> Result<&SelectingCharactersPlayer, GameError> {
33        self.players()
34            .iter()
35            .find(|p| p.name() == name)
36            .ok_or_else(|| GameError::InvalidPlayerName(name.to_owned()))
37    }
38
39    /// Gets a slice of all players in the lobby.
40    /// See [`Players::players`] for further information
41    pub fn players(&self) -> &[SelectingCharactersPlayer] {
42        self.players.players()
43    }
44
45    /// Gets the id of the current chairman
46    pub fn chairman_id(&self) -> PlayerId {
47        self.chairman
48    }
49
50    /// Gets the id of the player that's currently selecting a character
51    pub fn currently_selecting_id(&self) -> PlayerId {
52        (self.characters.applies_to_player() as u8).into()
53    }
54
55    /// Internally used function that checks whether a player with such an `id` exists, and whether
56    /// that player is the current player. If this is the case, a reference to the player is
57    /// returned.
58    fn player_as_current(&self, id: PlayerId) -> Result<&SelectingCharactersPlayer, GameError> {
59        let currently_selecting_id = self.currently_selecting_id();
60        match self.players.player(id) {
61            Ok(player) if player.id() == currently_selecting_id => Ok(player),
62            Ok(_) => Err(GameError::NotPlayersTurn),
63            Err(e) => Err(e),
64        }
65    }
66
67    /// Gets a list of selectable characters for the player with `id`, if it's their turn to select
68    /// a character next.
69    pub fn player_get_selectable_characters(
70        &self,
71        id: PlayerId,
72    ) -> Result<Vec<Character>, GameError> {
73        let _ = self.player_as_current(id)?;
74
75        self.characters
76            .peek()
77            .map(|pc| pc.characters)
78            .map_err(Into::into)
79    }
80
81    /// Gets the closed character for the player with `id` if they're chairman.
82    pub fn player_get_closed_character(&self, id: PlayerId) -> Result<Character, GameError> {
83        let _ = self.player_as_current(id)?;
84
85        match self.characters.peek()?.closed_character {
86            Some(closed_character) => Ok(closed_character),
87            None => Err(SelectingCharactersError::NotChairman.into()),
88        }
89    }
90
91    /// Allows player with `id` to select `character`, if it is their turn and if that character is
92    /// available to select. If they are the last player to select a character, a new [`GameState`]
93    /// is returned of type [`Round`].
94    pub(super) fn player_select_character(
95        &mut self,
96        id: PlayerId,
97        character: Character,
98    ) -> Result<Option<GameState>, GameError> {
99        let currently_selecting_id = self.currently_selecting_id();
100
101        match self.players.player_mut(id) {
102            Ok(p) if p.id() == currently_selecting_id => {
103                self.characters.pick(character)?;
104
105                p.select_character(character)?;
106
107                // Start round when no more characters can be picked
108                if self.characters.peek().is_err() {
109                    let current_player = self
110                        .players()
111                        .iter()
112                        .min_by(|p1, p2| p1.character().cmp(&p2.character()))
113                        .map(|p| p.id())
114                        .unwrap();
115                    // PANIC: This is safe because a game has to have at least four players to
116                    // start, and they cannot be removed
117
118                    let players = std::mem::take(&mut self.players);
119                    let assets = std::mem::take(&mut self.assets);
120                    let liabilities = std::mem::take(&mut self.liabilities);
121                    let markets = std::mem::take(&mut self.markets);
122                    let current_market = std::mem::take(&mut self.current_market);
123                    let current_events = std::mem::take(&mut self.current_events);
124                    let open_characters = self.characters.open_characters().to_vec();
125                    let fired_characters: Vec<Character> = vec![];
126                    let banker_target = None;
127                    let players = players
128                        .0
129                        .into_iter()
130                        .map(TryInto::try_into)
131                        .collect::<Result<_, _>>()?;
132
133                    let players = Players(players);
134
135                    let mut round = Round {
136                        current_player,
137                        players,
138                        assets,
139                        liabilities,
140                        markets,
141                        chairman: self.chairman,
142                        current_market,
143                        current_events,
144                        open_characters,
145                        fired_characters,
146                        banker_target,
147                        is_final_round: false,
148                    };
149
150                    round.players.player_mut(current_player)?.start_turn();
151
152                    Ok(Some(GameState::Round(round)))
153                } else {
154                    Ok(None)
155                }
156            }
157            Ok(_) => Err(GameError::NotPlayersTurn),
158            Err(e) => Err(e),
159        }
160    }
161
162    /// Gets the list of open characters, which are the characters nobody can select this round.
163    pub fn open_characters(&self) -> &[Character] {
164        self.characters.open_characters()
165    }
166
167    /// Gets a list of player ids that represent the order each player's turn is in. The chairman
168    /// id will always be the first id in this list, and ids will then count upward and loop back
169    /// around if necessary.
170    pub fn turn_order(&self) -> Vec<PlayerId> {
171        let start = usize::from(self.chairman) as u8;
172        let limit = self.players.len() as u8;
173        (start..limit).chain(0..start).map(Into::into).collect()
174    }
175
176    /// Get the current market
177    pub fn current_market(&self) -> &Market {
178        &self.current_market
179    }
180
181    /// Gets the [`PlayerInfo`] for each player, excluding the player that has the same id as `id`.
182    pub fn player_info(&self, id: PlayerId) -> Vec<PlayerInfo> {
183        self.players()
184            .iter()
185            .filter(|p| p.id() != id)
186            .map(|p| {
187                let mut info: PlayerInfo = p.into();
188                // Filter out the characters of players that have not had their turn yet
189                info.character = None;
190                info
191            })
192            .collect()
193    }
194
195    /// Sets a player as disconnected
196    pub fn leave(&mut self, id: PlayerId) -> Result<(), GameError> {
197        match self.players.player_mut(id) {
198            Ok(player) => {
199                player.set_is_human(false);
200                Ok(())
201            }
202            Err(e) => Err(e),
203        }
204    }
205    /// Allows a player to rejoin
206    pub fn rejoin(&mut self, id: PlayerId) -> Result<&SelectingCharactersPlayer, GameError> {
207        let player = self.players.player_mut(id)?;
208        if player.is_human() {
209            return Err(GameError::InvalidPlayerName(player.name().to_string()));
210        }
211        player.set_is_human(true);
212        Ok(player)
213    }
