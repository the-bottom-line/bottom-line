52    /// Pays the banker in the round with everything the player owns that are worth anything. This
53    /// means that this function ignores assets that are worth zero or negative cash in the current
54    /// market.
55    pub fn go_bankrupt_for_banker(
56        &mut self,
57        cash: u8,
58        banker: &mut BankerTargetPlayer,
59        market: Market,
60    ) -> Result<PayBankerPlayer, PayBankerError> {
61        let mut new_selected_cards: SelectedAssetsAndLiabilities = SelectedAssetsAndLiabilities {
62            sold_assets: vec![],
63            issued_liabilities: vec![],
64        };
65        for (index, asset) in self.assets.clone().into_iter().enumerate() {
66            if asset.market_value(&market) > 0 {
67                new_selected_cards.sold_assets.push(SoldAssetToPayBanker {
68                    asset_idx: index,
69                    market_value: asset.market_value(&market) as u8,
70                });
71            }
72        }
73        //get top 3 most valueble liabilities if player is CFO
74        if self.character == Character::CFO {
75            for (index, liability) in self
76                .hand
77                .clone()
78                .into_iter()
79                .filter(|l| l.is_right())
80                .enumerate()
81            {
82                if let Some(lib) = liability.right() {
83                    new_selected_cards
84                        .issued_liabilities
85                        .push(IssuedLiabilityToPayBanker {
86                            card_idx: index,
87                            liability: lib,
88                        });
89                }
90            }
91        }
92        let mut len = new_selected_cards.issued_liabilities.len();
93
94        if len > 3 {
95            len -= 3;
96        } else {
97            len = 0;
98        }
99        //remove smallest libilities if there are more as 3 in hand
100        for _i in 0..len {
101            let mut smallest_v = 0;
102            let mut index = 0;
103            for l in &new_selected_cards.issued_liabilities {
104                if smallest_v < l.liability.value {
105                    smallest_v = l.liability.value;
106                    index += 1;
107                }
108            }
109            new_selected_cards.issued_liabilities.remove(index);
110        }
111
112        // Sell assets and libilities for targeted player
113        let extra_asset_cash: u8 = new_selected_cards
114            .sold_assets
115            .iter()
116            .map(|s| s.market_value)
117            .sum();
118        let extra_liability_cash: u8 = new_selected_cards
119            .issued_liabilities
120            .iter()
121            .map(|l| l.liability.value)
122            .sum();
123        let mut asset_ids: Vec<usize> = new_selected_cards
124            .sold_assets
125            .iter()
126            .map(|s| s.asset_idx)
127            .collect();
128        asset_ids.sort();
129        for id in asset_ids.iter().rev() {
130            self.assets.remove(*id);
131        }
132
133        let mut liability_ids: Vec<usize> = new_selected_cards
134            .issued_liabilities
135            .iter()
136            .map(|l| l.card_idx)
137            .collect();
138        liability_ids.sort();
139        for id in liability_ids.iter().rev() {
140            self.hand.remove(*id);
141            self.liabilities_to_play -= 1;
142        }
143        let total_available_cash = extra_asset_cash + extra_liability_cash + self.cash;
144        if total_available_cash < cash {
145            //TODO Pay banker the maximum amount target can affort after selling
146            banker.cash += total_available_cash;
147            self.cash = 0;
148
149            Ok(PayBankerPlayer {
150                paid_amount: total_available_cash,
151                new_banker_cash: banker.cash,
152                new_target_cash: self.cash,
153                target_id: self.id,
154                banker_id: banker.id,
155                selected_cards: new_selected_cards.clone(),
156            })
157        } else {
158            Err(PayBankerError::NotRightCashAmount {
159                expected: total_available_cash,
160                got: cash,
161            })
162        }
163    }

165    /// Pays the banker in the round the requested amount of gold
166    pub fn pay_banker(
167        &mut self,
168        cash: u8,
169        selected_assets: &HashMap<usize, u8>,
170        selected_liabilities: &HashMap<usize, u8>,
171        banker: &mut BankerTargetPlayer,
172    ) -> Result<PayBankerPlayer, PayBankerError> {
173        let extra_asset_cash = selected_assets.values().sum::<u8>();
174        let extra_liability_cash = selected_liabilities.values().sum::<u8>();
175
176        if self.cash + extra_asset_cash + extra_liability_cash >= cash {
177            banker.cash += cash;
178            self.cash += extra_asset_cash + extra_liability_cash;
179            self.cash -= cash;
180
181            // TODO: reuse in `create_select_assets_liabilities` somehow
182            let sold_assets = selected_assets
183                .iter()
184                .map(|(&asset_idx, &market_value)| SoldAssetToPayBanker {
185                    asset_idx,
186                    market_value,
187                })
188                .collect::<Vec<_>>();
189
190            let issued_liabilities = selected_liabilities
191                .iter()
192                .map(|(&card_idx, _)| {
193                    // TODO: verify legitimacy of unwrapping here
194                    let liability = self.hand.get(card_idx).unwrap().clone().right().unwrap();
195                    IssuedLiabilityToPayBanker {
196                        card_idx,
197                        liability,
198                    }
199                })
200                .collect::<Vec<_>>();
201
202            let selected_cards = SelectedAssetsAndLiabilities {
203                sold_assets,
204                issued_liabilities,
205            };
206
207            let mut asset_idxs = selected_assets.iter().map(|a| *a.0).collect::<Vec<_>>();
208
209            asset_idxs.sort();
210
211            for asset_idx in asset_idxs.iter().rev() {
212                // TODO: figure out if this can have invalid indices
213                self.assets.remove(*asset_idx);
214            }
215
216            let mut liability_idxs = selected_liabilities
217                .iter()
218                .map(|l| *l.0)
219                .collect::<Vec<_>>();
220
221            liability_idxs.sort();
222
223            for card_idx in liability_idxs.iter().rev() {
224                self.hand.remove(*card_idx);
225                self.liabilities_to_play -= 1;
226            }
227
228            Ok(PayBankerPlayer {
229                paid_amount: cash,
230                new_banker_cash: banker.cash,
231                new_target_cash: self.cash,
232                target_id: self.id,
233                banker_id: banker.id,
234                selected_cards,
235            })
236        } else {
237            Err(PayBankerError::NotEnoughCash)
238        }
239    }

241    /// Select an asset to divest later when paying the banker
242    pub fn select_divest_asset(
243        &mut self,
244        asset_id: usize,
245        market: &Market,
246        selected_assets: &mut HashMap<usize, u8>,
247    ) -> Result<&Asset, BankerTargetSelectError> {
248        if let Some(asset) = self.assets.get(asset_id) {
249            if let Entry::Vacant(entry) = selected_assets.entry(asset_id) {
250                let market_value = asset.market_value(market);
251                if market_value > 0 {
252                    entry.insert(market_value as u8);
253                    Ok(asset)
254                } else {
255                    Err(BankerTargetSelectError::AssetValueToLow)
256                }
257            } else {
258                Err(BankerTargetSelectError::AssetAlreadySelected)
259            }
260        } else {
261            // TODO: use GameError::InvalidAssetIndex or self.asset(asset_idx)
262            Err(BankerTargetSelectError::InvalidAssetId(asset_id as u8))
263        }
264    }

266    /// Unselect an asset to remove it from divest asset list when paying the banker
267    pub fn unselect_divest_asset(
268        &mut self,
269        asset_id: usize,
270        selected_assets: &mut HashMap<usize, u8>,
271    ) -> Result<&Asset, BankerTargetSelectError> {
272        if let Some(asset) = self.assets.get(asset_id) {
273            if let Some(_market_value) = selected_assets.remove(&asset_id) {
274                Ok(asset)
275            } else {
276                Err(BankerTargetSelectError::AssetNotSelected)
277            }
278        } else {
279            // TODO: use GameError::InvalidAssetIndex or self.asset(asset_idx)
280            Err(BankerTargetSelectError::InvalidAssetId(asset_id as u8))
281        }
282    }

284    /// Select a liability from this player's hand and adds it to the issue liability list when
285    /// paying the banker
286    pub fn select_issue_liability(
287        &mut self,
288        card_idx: usize,
289        selected_liabilities: &mut HashMap<usize, u8>,
290    ) -> Result<&Liability, BankerTargetSelectError> {
291        if self.character == Character::CFO {
292            if let Some(Either::Right(liability)) = self.hand.get(card_idx) {
293                let playable_liabilities = Character::CFO.playable_liabilities() as usize;
294                if selected_liabilities.len() < playable_liabilities {
295                    if let Entry::Vacant(entry) = selected_liabilities.entry(card_idx) {
296                        entry.insert(liability.value);
297                        Ok(liability)
298                    } else {
299                        Err(BankerTargetSelectError::LiabilityAlreadySelected)
300                    }
301                } else {
302                    Err(BankerTargetSelectError::AlreadySelected3Liabilities)
303                }
304            } else {
305                Err(BankerTargetSelectError::InvalidLiabilityId(card_idx as u8))
306            }
307        } else {
308            Err(BankerTargetSelectError::NotCFO)
309        }
310    }
