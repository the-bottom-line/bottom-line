26/// Representation of an asset card. Each asset has a gold and a silver value, as well as an
27/// associated color. Some cards alse have an [`AssetPowerup`].
28#[cfg_attr(feature = "ts", derive(TS))]
29#[cfg_attr(feature = "ts", ts(rename = "AssetCard"))]
30#[cfg_attr(feature = "ts", ts(export_to = crate::SHARED_TS_DIR))]
31#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
32pub struct Asset {
33    /// Title of the asset card.
34    pub title: String,
35    /// The gold value of the asset.
36    pub gold_value: u8,
37    /// The silver value of the asset.
38    pub silver_value: u8,
39    /// The color of the asset.
40    pub color: Color,
41    /// Whether or not this asset has an [`AssetPowerup`].
42    pub ability: Option<AssetPowerup>,
43    /// Url containing the relative location of the card in the assets folder
44    pub image_front_url: String,
45    /// Url containing the relative location of the back of the card in the assets folder
46    pub image_back_url: Arc<String>,
47}
48
49impl Asset {
50    /// Gets the current value of the asset based on the given market condition. Note that this
51    /// value can be negative.
52    pub fn market_value(&self, market: &Market) -> i8 {
53        let mul: i8 = match market.color_condition(self.color) {
54            MarketCondition::Plus => 1,
55            MarketCondition::Minus => -1,
56            MarketCondition::Zero => 0,
57        };
58        self.gold_value as i8 + self.silver_value as i8 * mul
59    }
60
61    /// Calculates what it costs to divest this asset based on the current market. Note that this
62    /// number as zero at its lowest.
63    pub fn divest_cost(&self, market: &Market) -> u8 {
64        let mv = self.market_value(market);
65
66        // match mv {
67        //     ..=1 => 0,
68        //     n => n as u8 - 1
69        // }
70        // mv.max(1) as u8 - 1
71        if mv <= 1 { 0 } else { (mv - 1) as u8 }
72    }
73}

92/// Representation of a liability card. Each liability has an associated gold value as well as a
93/// [`LiabilityType`], which determines how expensive it is to issue this liability.
94#[cfg_attr(feature = "ts", derive(TS))]

97#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
98pub struct Liability {
99    /// Gold value of this liability
100    pub value: u8,
101    /// The card's [`LiabilityType`], which determines how expensive it is to issue this liability.
102    pub rfr_type: LiabilityType,
103    /// Url containing the relative location of the card in the assets folder.
104    pub image_front_url: String,
105    /// Url containing the relative location of the back of the card in the assets folder.
106    pub image_back_url: Arc<String>,
107}
108
109impl Liability {
110    /// Gets the associated rfr% for this liability. This can either be 1, 2 or 3.
111    pub fn rfr_percentage(&self) -> u8 {
112        match self.rfr_type {
113            LiabilityType::TradeCredit => 1,
114            LiabilityType::BankLoan => 2,
115            LiabilityType::Bonds => 3,
116        }
117    }
118}

147/// Trait that should be implemented for each player type to be able to transform its internal data
148/// into publicly displayable [`PlayerInfo`].
149pub trait GetPlayerInfo {
150    /// Gets the publicly available info of this particular player.
151    fn info(&self) -> PlayerInfo;
152}
153
154impl<T> GetPlayerInfo for T
155where
156    for<'a> PlayerInfo: From<&'a T>,
157{
158    fn info(&self) -> PlayerInfo {
159        PlayerInfo::from(self)
160    }
161}

163/// Publicly available information for each player. This contains information that you would be able
164/// to see from another player if you were looking at what they have on the table. You cannot see
165/// their hand, but you can see the amount of asset cards and liability cards they have.
166#[cfg_attr(feature = "ts", derive(TS))]
167#[cfg_attr(feature = "ts", ts(export_to = crate::SHARED_TS_DIR))]
168#[derive(Debug, Clone, Serialize, Deserialize)]
169pub struct PlayerInfo {
170    /// The name of the player.
171    pub name: String,
172    /// The id of the player.
173    pub id: PlayerId,
174    /// The hand of the player, represented as different [`CardType`]s.
175    pub hand: Vec<CardType>,
176    /// The assets this player has bought.
177    pub assets: Vec<Asset>,
178    /// The liabilities this player has issued.
179    pub liabilities: Vec<Liability>,
180    /// The amount of cash this player has.
181    pub cash: u8,
182    /// The character this player has chosen, if applicable.
183    pub character: Option<Character>,
184    /// This player is controlled by a human
185    pub is_human: bool,
186}
187
188impl PlayerInfo {
189    fn hand(hand: &[Either<Asset, Liability>]) -> Vec<CardType> {
190        hand.iter()
191            .map(|e| match e {
192                Either::Left(_) => CardType::Asset,
193                Either::Right(_) => CardType::Liability,
194            })
195            .collect()
196    }
197}
198
199impl Default for PlayerInfo {
200    fn default() -> Self {
201        Self {
202            name: Default::default(),
203            id: PlayerId(0),
204            hand: Default::default(),
205            assets: Default::default(),
206            liabilities: Default::default(),
207            cash: Default::default(),
208            character: Default::default(),
209            is_human: Default::default(),
210        }
211    }
212}

301/// An enum containing all characters currently in the game in the order in which they are called.
302#[derive(Debug, Copy, Clone, Serialize, Deserialize, PartialEq, Eq, Hash, PartialOrd, Ord)]
303#[cfg_attr(feature = "ts", derive(TS))]
304#[cfg_attr(feature = "ts", ts(rename = "CharacterType"))]
305#[cfg_attr(feature = "ts", ts(export_to = crate::SHARED_TS_DIR))]
306pub enum Character {
307    /// This character can fire any other character during their turn, excluding the
308    /// [`Banker`](Character::Banker) and the [`Regulator`](Character::Regulator).
309    Shareholder,
310    /// This character can terminate the credit of any other character except for the
311    /// [`Shareholder`](Character::Shareholder) and the [`Regulator`](Character::Regulator). This
312    /// means that at the start of their turn they pay one gold to the banker plus one gold per
313    /// unique color of assets they own.
314    Banker,
315    /// This character can choose to swap either their hand with the hand of any other player, or to
316    /// swap any number of cards in their hand with random cards of the same type in the deck.
317    Regulator,
318    /// This character can buy up to three assets of any color during their turn.
319    CEO,
320    /// This character can issue up to three liabilities each turn. Alternatively, they can also
321    /// choose to redeem liabilities, where they pay the liability's gold in cash to get it off
322    /// their balance sheet.
323    CFO,
324    /// This character can buy up to two red or green assets.
325    CSO,
326    /// This character is allowed to draw six cards, giving back two.
327    HeadRnD,
328    /// This character can force any player except for the [`CSO`](Character::CSO) to divest one of
329    /// their assets at market value minus one. This value cannot be negative and is paid for by
330    /// this character.
331    Stakeholder,
332}
333
334impl Character {
335    /// A list of all characters in this enum in the order they are called.
336    pub const CHARACTERS: [Character; 8] = [
337        Self::Shareholder,
338        Self::Banker,
339        Self::Regulator,
340        Self::CEO,
341        Self::CFO,
342        Self::CSO,
343        Self::HeadRnD,
344        Self::Stakeholder,
345    ];
346
347    /// Gets the [`Color`] which some characters have associated with them.
348    pub fn color(&self) -> Option<Color> {
349        use Color::*;
350
351        match self {
352            Self::Shareholder => None,
353            Self::Banker => None,
354            Self::Regulator => None,
355            Self::CEO => Some(Yellow),
356            Self::CFO => Some(Blue),
357            Self::CSO => Some(Green),
358            Self::HeadRnD => Some(Purple),
359            Self::Stakeholder => Some(Red),
360        }
361    }
362
363    /// Gets the character who is called after this character
364    pub fn next(&self) -> Option<Self> {
365        use Character::*;
366
367        match self {
368            Shareholder => Some(Banker),
369            Banker => Some(Regulator),
370            Regulator => Some(CEO),
371            CEO => Some(CFO),
372            CFO => Some(CSO),
373            CSO => Some(HeadRnD),
374            HeadRnD => Some(Stakeholder),
375            Stakeholder => None,
376        }
377    }
378
379    /// Gets the character that is called first from a list of characters
380    pub fn first(characters: &[Self]) -> Option<Self> {
381        characters.iter().min().copied()
382    }
383
384    /// Gets this character's [`PlayableAssets`], which is a representation of how many assets of
385    /// each color this character can buy this round.
386    pub fn playable_assets(&self) -> PlayableAssets {
387        match self {
388            Self::CEO => PlayableAssets {
389                total: 3,
390                ..Default::default()
391            },
392            Self::CSO => PlayableAssets {
393                total: 2,
394                red_cost: 1,
395                green_cost: 1,
396                purple_cost: 2,
397                yellow_cost: 2,
398                blue_cost: 2,
399            },
400            _ => PlayableAssets::default(),
401        }
402    }
403
404    /// Gets the amount of liabilities this character can issue.
405    pub fn playable_liabilities(&self) -> u8 {
406        match self {
407            Self::CFO => 3,
408            _ => 1,
409        }
410    }
411
412    /// Gets the amount of cards this character is allowed to draw.
413    pub fn draws_n_cards(&self) -> u8 {
414        // TODO: fix head rnd ability when ready
415        match self {
416            Self::HeadRnD => 6,
417            _ => 3,
418        }
419    }
420
421    /// Returns `true` if this character is allowed to redeem liabilities
422    pub fn can_redeem_liabilities(&self) -> bool {
423        matches!(self, Self::CFO)
424    }
425
426    /// Returns `true` if this character is allowed to fire other characters.
427    pub fn can_fire_characters(&self) -> bool {
428        matches!(self, Self::Shareholder)
429    }
430
431    /// Returns `true` if this character can force other players to divest an asset.
432    pub fn can_force_others_to_divest(&self) -> bool {
433        matches!(self, Self::Stakeholder)
434    }
435
436    /// Returns `true` if this character can be fired
437    pub fn can_be_fired(&self) -> bool {
438        matches!(
439            self,
440            Self::CEO | Self::CSO | Self::CFO | Self::HeadRnD | Self::Stakeholder
441        )
442    }
443
444    /// Returns true if this character can be forced to divest.
445    pub fn can_be_forced_to_divest(&self) -> bool {
446        !matches!(self, Self::CSO)
447    }
448}

450/// a representation of how many assets of each color a certain player is allowed to buy this round.
451#[cfg_attr(feature = "ts", derive(TS))]
452#[cfg_attr(feature = "ts", ts(export_to = crate::SHARED_TS_DIR))]
453#[derive(Debug, Copy, Clone, Serialize, Deserialize, PartialEq)]
454pub struct PlayableAssets {
455    total: u8,
456    red_cost: u8,
457    green_cost: u8,
458    purple_cost: u8,
459    yellow_cost: u8,
460    blue_cost: u8,
461}
462
463impl PlayableAssets {
464    /// The total unit value of assets a player can buy
465    pub fn total(&self) -> u8 {
466        self.total
467    }
468
469    /// The unit cost of buying an asset of a certain color.
470    pub fn color_cost(&self, color: Color) -> u8 {
471        let cost = match color {
472            Color::Red => self.red_cost,
473            Color::Green => self.green_cost,
474            Color::Purple => self.purple_cost,
475            Color::Yellow => self.yellow_cost,
476            Color::Blue => self.blue_cost,
477        };
478
479        debug_assert!(cost > 0);
480        debug_assert_eq!(self.total % cost, 0);
481
482        cost
483    }
484}
