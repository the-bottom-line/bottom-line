3pub mod serde_asset_liability {
4    //! Includes a tagged wrapper around `Either<Asset, Liability>` to make the serialized json
5    //! nicer to interact with.
6
7    use either::Either;
8    use serde::{Deserialize, Deserializer, Serialize, Serializer};
9
10    #[cfg(feature = "ts")]
11    use ts_rs::TS;
12
13    use crate::player::{Asset, Liability};
14
15    /// A nicer tagged representation of `Either<Asset, Liability>` which looks much better when
16    /// serialized.
17    #[cfg_attr(feature = "ts", derive(TS))]
18    #[cfg_attr(feature = "ts", ts(export_to = crate::SHARED_TS_DIR))]
19    #[derive(Debug, Clone, Serialize, Deserialize)]
20    #[serde(tag = "card_type")]
21    pub enum EitherAssetLiability {
22        /// `"card_type": "asset"
23        #[serde(rename = "asset")]
24        Asset(Asset),
25        /// `"card_type": "liability"
26        #[serde(rename = "liability")]
27        Liability(Liability),
28    }
29
30    impl From<EitherAssetLiability> for Either<Asset, Liability> {
31        fn from(w: EitherAssetLiability) -> Self {
32            match w {
33                EitherAssetLiability::Asset(a) => Either::Left(a),
34                EitherAssetLiability::Liability(l) => Either::Right(l),
35            }
36        }
37    }
38
39    impl From<Either<Asset, Liability>> for EitherAssetLiability {
40        fn from(e: Either<Asset, Liability>) -> Self {
41            match e {
42                Either::Left(a) => EitherAssetLiability::Asset(a),
43                Either::Right(l) => EitherAssetLiability::Liability(l),
44            }
45        }
46    }
47
48    pub mod value {
49        //! Serde module containing an implementation of `serialize` and `deserialize` which can be
50        //! used to serialize `Either<Asset, Liability>` to a nicer tagged representation.
51
52        use super::*;
53
54        /// Serializes an `Either<Asset, Liability>` to a nicer tagged representation.
55        pub fn serialize<S>(
56            value: &Either<Asset, Liability>,
57            serializer: S,
58        ) -> Result<S::Ok, S::Error>
59        where
60            S: Serializer,
61        {
62            EitherAssetLiability::from(value.clone()).serialize(serializer)
63        }
64
65        /// Deserializes the nicer tagged representation back into `Either<Asset, Liability>`
66        pub fn deserialize<'de, D>(deserializer: D) -> Result<Either<Asset, Liability>, D::Error>
67        where
68            D: Deserializer<'de>,
69        {
70            EitherAssetLiability::deserialize(deserializer).map(Either::from)
71        }
72    }
73
74    pub mod vec {
75        //! Module intended to be used with `#[serde(with)]` containing an implementation of
76        //! `serialize` and `deserialize` which can be used to serialize
77        //! `Vec<Either<Asset, Liability>>` to a nicer tagged representation.
78
79        use super::*;
80
81        /// Serializes a `Vec<Either<Asset, Liability>>` into a nicer tagged representation
82        pub fn serialize<S>(
83            value: &[Either<Asset, Liability>],
84            serializer: S,
85        ) -> Result<S::Ok, S::Error>
86        where
87            S: Serializer,
88        {
89            let mapped = value
90                .iter()
91                .cloned()
92                .map(EitherAssetLiability::from)
93                .collect::<Vec<_>>();
94
95            mapped.serialize(serializer)
96        }
97
98        /// Deserializes the list of nicer tagged representations back into a
99        /// `Vec<Either<Asset, Liability>>`
100        pub fn deserialize<'de, D>(
101            deserializer: D,
102        ) -> Result<Vec<Either<Asset, Liability>>, D::Error>
103        where
104            D: Deserializer<'de>,
105        {
106            let intermediate = Vec::<EitherAssetLiability>::deserialize(deserializer)?;
107            Ok(intermediate.into_iter().map(Either::from).collect())
108        }
109    }
110}
