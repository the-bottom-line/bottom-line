29/// Requests that are sent from the frontend, to be handled by the backend.
30#[cfg_attr(feature = "ts", derive(TS))]
31#[cfg_attr(feature = "ts", ts(export_to = game::SHARED_TS_DIR))]
32#[cfg_attr(feature = "ts", ts(export))]
33#[derive(Clone, Debug, Serialize, Deserialize)]
34#[serde(tag = "action", content = "data")]
35pub enum FrontendRequest {
36    /// Tries to start the game.
37    StartGame,
38    /// Tries to select a character for this player.
39    SelectCharacter {
40        /// The character the player wants to select.
41        character: Character,
42    },
43    /// Tries to draw a card for this player.
44    DrawCard {
45        /// The [`CardType`] the player wants to draw.
46        card_type: CardType,
47    },
48    /// Tries to put back a card for this player.
49    PutBackCard {
50        /// The index of the card this player is trying to put back.
51        card_idx: usize,
52    },
53    /// Tries to buy an asset for this player.
54    BuyAsset {
55        /// The index of the card the player wants to buy.
56        card_idx: usize,
57    },
58    /// Tries to issue a liability for this player.
59    IssueLiability {
60        /// The index of the card the player wants to issue.
61        card_idx: usize,
62    },
63    /// Tries to redeem a liability for this player.
64    RedeemLiability {
65        /// The index of the issued liability the player wanst to redeem.
66        liability_idx: usize,
67    },
68    /// Tries to use the ability for this player.
69    UseAbility,
70    /// Get characters bonus gold only once per turn,
71    GetBonusCash,
72    /// Tries to fire a particular character by this player.
73    FireCharacter {
74        /// The character that is to be fired.
75        character: Character,
76    },
77    /// Tries toterminate credit from particular character by this player.
78    TerminateCreditCharacter {
79        /// The character who's credit line will be terminated.
80        character: Character,
81    },
82    /// Tries to select an asset to sell to pay off the banker
83    SelectAssetToDivest {
84        /// The index of the asset the player wants to select to pay off the banker.
85        asset_id: usize,
86    },
87    /// Tries to unselect an asset they were set to sell to pay off the banker.
88    UnselectAssetToDivest {
89        /// The index of the asset the player wants to unselect.
90        asset_id: usize,
91    },
92    /// Tries to select a liability from the player's hand to issue to pay off the banker
93    SelectLiabilityToIssue {
94        /// The index of the liability in the player's hand to select to issue to pay off the
95        /// banker.
96        liability_id: usize,
97    },
98    /// Tries to unselect a liability they were set to issue to pay off the banker.
99    UnselectLiabilityToIssue {
100        /// The index of the liability in the player's hand the player wants to unselect.
101        liability_id: usize,
102    },
103    /// Tries to send cash to the banker when player is targeted
104    PayBanker {
105        /// The amount of cash to pay
106        cash: u8,
107    },
108    /// Tries to swap a list of card indices with the deck for this player.
109    SwapWithDeck {
110        /// The list of card indices to be swapped with the deck.
111        card_idxs: Vec<usize>,
112    },
113    /// Tries to swap hands of this player with another player.
114    SwapWithPlayer {
115        /// The id of the player which is to be swapped with.
116        target_player_id: PlayerId,
117    },
118    /// Tries to force another player to divest one of their assets at market value minus one, which
119    /// is to be paid by this player.
120    DivestAsset {
121        /// The id of the player which would be forced to divest their asset.
122        target_player_id: PlayerId,
123        /// The index of the asset that is to be divested.
124        card_idx: usize,
125    },
126    /// Tries to end the turn of this player.
127    EndTurn,
128    /// Request a resync packet containing the current gamestate
129    Resync,
130    /// Tries to turn minus into zero or zero into plus for the player's market at the end of the
131    /// game. Related to [`AssetPowerup::MinusIntoPlus`](game::player::AssetPowerup::MinusIntoPlus).
132    MinusIntoPlus {
133        /// The color to change the minus from.
134        color: Color,
135    },
136    /// Tries to turn the silver of a particular asset into gold. Related to
137    /// [`AssetPowerup::SilverIntoGold`](game::player::AssetPowerup::MinusIntoPlus).
138    SilverIntoGold {
139        /// The index of the asset to change silver into gold from.
140        asset_idx: usize,
141    },
142    /// Tries to change the color of any bought asset into another color. Related to
143    /// [`AssetPowerup::ChangeAssetColor`](game::player::AssetPowerup::ChangeAssetColor).
144    ChangeAssetColor {
145        /// The index of the asset to change color from.
146        asset_idx: usize,
147        /// The new color of the asset.
148        color: Color,
149    },
150    /// Tries to confirm the usage of a asset ability.
151    ConfirmAssetAbility {
152        /// The index of the asset which ability was used.
153        asset_idx: usize,
154    },
155}

157/// A response type that a player receives after performing an action. Can either be an error or
158/// a confirmation that the action was succesful, including the data needed to update the UI
159/// accordingly.
160#[cfg_attr(feature = "ts", derive(TS))]
161#[cfg_attr(feature = "ts", ts(export_to = game::SHARED_TS_DIR))]
162#[cfg_attr(feature = "ts", ts(export))]
163#[derive(Debug, Serialize, Deserialize)]
164#[serde(tag = "action", content = "data")]
165pub enum DirectResponse {
166    /// An error returned when the action was not succesful.
167    Error {
168        /// The error message.
169        message: String,
170        /// The error type.
171        source: ResponseError,
172    },
173    /// Confirmation that this player started the game.
174    YouStartedGame,
175    /// Confirmation that this player selected a character.
176    YouSelectedCharacter {
177        /// The character this player selected.
178        character: Character,
179    },
180    /// Confirmation that this player fired a character.
181    YouFiredCharacter {
182        /// The character that was fired.
183        character: Character,
184    },
185    /// Confirmation that this character's credit line is terminated.
186    YouTerminateCreditCharacter {
187        /// The character who's credit line was terminated.
188        character: Character,
189    },
190    /// Confirmation that you paid some gold to the banker.
191    YouPaidBanker {
192        /// The id of the player who is the banker.
193        banker_id: PlayerId,
194        /// The new cash balance of the banker.
195        new_banker_cash: u8,
196        /// The new cash balance of the player that was targeted by the banker.
197        your_new_cash: u8,
198        /// The amount of gold paid.
199        paid_amount: u8,
200        /// A list of assets to be sold to pay off the banker.
201        sold_assets: Vec<SoldAssetToPayBanker>,
202        /// A list of liabilities to be issued to pay off the banker.
203        issued_liabilities: Vec<IssuedLiabilityToPayBanker>,
204    },
205    /// Confirmation that you selected someone as target as the banker.
206    YouSelectCardBankerTarget {
207        /// A list of assets to be sold to pay off the banker.
208        assets: Vec<SoldAssetToPayBanker>,
209        /// A list of liabilities to be issued to pay off the banker.
210        liabilities: Vec<IssuedLiabilityToPayBanker>,
211    },
212    /// Confirmation that this player was succesful in getting regulator options
213    YouRegulatorOptions {
214        /// The options this player has to swap with other players.
215        options: Vec<RegulatorSwapPlayer>,
216        /// Always [`Character::Regulator`]
217        character: Character,
218        /// A string containing information about what this character is allowed to do.
219        perk: String,
220    },
221    /// Confirmation that this player can proceed swapping with the deck.
222    YouSwapDeck {
223        /// The amount of cards this player may draw from the deck.
224        cards_to_draw: usize,
225    },
226    /// Confirmation that this player was succesful in swapping with a player.
227    YouSwapPlayer {
228        /// The new hand of the player.
229        #[cfg_attr(
230            feature = "ts",
231            ts(as = "Vec<serde_asset_liability::EitherAssetLiability>")
232        )]
233        #[serde(with = "serde_asset_liability::vec")]
234        new_cards: Vec<Either<Asset, Liability>>,
235        /// The id of the player you swapped cards with
236        target_player_id: PlayerId,
237    },
238    /// Confirmation that this player is now forcing another player to divest.
239    YouAreDivesting {
240        /// A list of cards for each player, which can either be or not be divested.
241        options: Vec<DivestPlayer>,
242        /// Always [`Character::Stakeholder`]
243        character: Character,
244        /// A string containing information about what this character is allowed to do.
245        perk: String,
246    },
247    /// Confirmation that this player drew a card.
248    YouDrewCard {
249        /// The card that was drawn
250        #[cfg_attr(feature = "ts", ts(as = "serde_asset_liability::EitherAssetLiability"))]
251        #[serde(with = "serde_asset_liability::value")]
252        card: Either<Asset, Liability>,
253        /// Whether this player can draw another card.
254        can_draw_cards: bool,
255        /// Whether this player should still give back any cards.
256        can_give_back_cards: bool,
257    },
258    /// Confirmation that this player put back a card.
259    YouPutBackCard {
260        /// The index of the card this player put back.
261        card_idx: usize,
262        /// Whether this player can draw another card.
263        can_draw_cards: bool,
264        /// Whether this player should still give back any cards.
265        can_give_back_cards: bool,
266    },
267    /// Confirmation that this player is trying to use their character ability.
268    YouCharacterAbility {
269        /// The character of the player.
270        character: Character,
271        /// A string containing information about what this player is allowed to do.
272        perk: String,
273    },
274    /// Confirmation you received your bonus cash and how much.
275    YouBonusCash {
276        /// The amount of cash received
277        cash: u8,
278    },
279    /// Confirmation that this player bought an asset.
280    YouBoughtAsset {
281        /// The asset this player bought.
282        asset: Asset,
283        /// The index of the asset in the player's hand this player bought.
284        card_idx: usize,
285        /// If the market changed, a list of events and a new market is returned.
286        market_change: Option<MarketChange>,
287    },
288    /// Confirmation that this player issued a liability.
289    YouIssuedLiability {
290        /// The liability the player issued.
291        liability: Liability,
292        /// The index of the liability in the player's hand this player issued.
293        card_idx: usize,
294    },
295    /// Confirmation that this player
296    YouAreFiringSomeone {
297        /// The list of available characters to fire.
298        characters: Vec<Character>,
299        /// Always [`Character::Shareholder`]
300        character: Character,
301        /// A string containing information on what this character is allowed to do.
302        perk: String,
303    },
304    /// Confirmation that this player divested an asset of another player.
305    YouDivestedAnAsset {
306        /// The id of the player who is forced to divest one of their assets.
307        target_id: PlayerId,
308        /// The index of the asset they are forced to divest.
309        asset_idx: usize,
310        /// The amount of gold it cost to divest this asset.
311        gold_cost: u8,
312    },
313    /// Confirmation that this player is terminating the credit of another player.
314    YouAreTerminatingSomeone {
315        /// A list of characters whose credit can be terminated.
316        characters: Vec<Character>,
317        /// Always [`Character::Banker`]
318        character: Character,
319        /// A string containing information on what this character is allowed to do.
320        perk: String,
321    },
322    /// Confirmation that this player redeemed a liability.
323    YouRedeemedLiability {
324        /// The index of the liability that was redeemed.
325        liability_idx: usize,
326    },
327    /// Confirmation that this player ended their turn.
328    YouEndedTurn,
329
330    /// Confirmation that you joined the lobby.
331    YouJoinedGame {
332        /// Your connected username.
333        username: String,
334        /// The channel you're connected to.
335        channel: String,
336    },
337    /// Inform the client that they are rejoining
338    YouRejoined,
339
340    /// Deliver data to the client to sync them back to the game state
341    YouResynced {
342        /// This player's personal id.
343        id: PlayerId,
344        /// The amount of cash this player has.
345        cash: u8,
346        /// The player's hand.
347        #[cfg_attr(
348            feature = "ts",
349            ts(as = "Vec<serde_asset_liability::EitherAssetLiability>")
350        )]
351        #[serde(with = "serde_asset_liability::vec")]
352        hand: Vec<Either<Asset, Liability>>,
353        /// The assets already played by the player
354        assets: Vec<Asset>,
355        /// The liabilities already played by the player
356        liabilities: Vec<Liability>,
357        /// Public info about every other player.
358        player_info: Vec<PlayerInfo>,
359        /// The current market.
360        market: Market,
361        /// A response containing the current gamestate
362        phase: ResyncData,
363    },
364    /// Confirms that this player changed one of their market colors.
365    YouMinusedIntoPlus {
366        /// The market color that was changed,
367        color: Color,
368        /// The new market for this player.
369        new_market: Market,
370        /// The updated player score.
371        new_score: f64,
372    },
373    /// Confirms that this player changed the silver of one of their cards into gold.
374    YouSilveredIntoGold {
375        /// The data of the old asset that should be updated. When no card was selected before this
376        /// action, this value will be `None`.
377        old_asset_data: Option<SilverIntoGoldData>,
378        /// The data of the new asset that should be updated. When a card was deselected, this value
379        /// will be `None`.
380        new_asset_data: Option<SilverIntoGoldData>,
381        /// The updated player score.
382        new_score: f64,
383    },
384    /// Confirms that this player changed the color of one of their assets.
385    YouChangedAssetColor {
386        /// The data of the old asset that should be updated. When no card was selected before this
387        /// action, this value will be `None`.
388        old_asset_data: Option<ChangeAssetColorData>,
389        /// The data of the new asset that should be updated. When a card was deselected, this value
390        /// will be `None`.
391        new_asset_data: Option<ChangeAssetColorData>,
392        /// The updated player score.
393        new_score: f64,
394    },
395    /// Confirms that this player confirmed an asset ability's choice.
396    YouConfirmedAssetAbility {
397        /// The asset the player confirmed their choice for.
398        asset_idx: usize,
399    },
400}

402impl From<ResponseError> for DirectResponse {
403    fn from(error: ResponseError) -> Self {
404        DirectResponse::Error {
405            message: error.to_string(),
406            source: error,
407        }
408    }
409}

411impl From<GameError> for DirectResponse {
412    fn from(error: GameError) -> Self {
413        DirectResponse::Error {
414            message: error.to_string(),
415            source: error.into(),
416        }
417    }

420/// A response type that is meant for every other player when one player performs an action.

660        /// The data of the new asset that should be updated. When a card was deselected, this value
661        /// will be `None`.
662        new_asset_data: Option<SilverIntoGoldData>,
663        /// The updated player score.
664        new_score: f64,
665    },
666    /// Confirms that a player changed the color of one of their assets.
667    ChangedAssetColor {
668        /// The id of the player which changed the color of one of their assets.
669        player_id: PlayerId,
670        /// The data of the old asset that should be updated. When no card was selected before this

672        old_asset_data: Option<ChangeAssetColorData>,

678    },
679    /// Confirms that a player confirmed an asset ability's choice.

683        /// The asset the player confirmed their choice for.
684        asset_idx: usize,
685    },
